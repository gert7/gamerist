def trickle_down_score (lastr, lastu)
  if(lastu < self.amount)
    self.balance_u = 0
    self.balance_r = self.amount - lastu # remaining
  else
    self.balance_u = lastu - self.amount
    self.balance_r = lastr
  end
end
