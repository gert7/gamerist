def country(code)
  case code
  when :EST
    return {
      vat: 0.20,
      compensation: 0.10, # compensate for 10% of VAT
      paypalcurrency: :EUR,
      masspaycurrency: :EUR,
      masspayrate: 0.02,
      masspayfallout: 6.0
    }
  else
    return {
      vat: 0.20,
      compensation: 0.10,
      paypalcurrency: :EUR,
      masspaycurrency: :EUR,
      masspayrate: 0.02,
      masspayfallout: 35.0
    }
  end
end
